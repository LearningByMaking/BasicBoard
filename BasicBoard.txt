constants [[rpacket $1fc0]] ;this constant defined the memory address where packets are written on the MSP430

 to ul-power
	 init-pins ;this initializes the LED pins
	 redon 
	 wait 10 
	 redoff 
 end
 
 
 
 to ul-init
	 yellowon 
	 wait 10
	 yellowoff
 end

 
 
 to ul-go
 	 greenon
	 wait 10
	 greenoff

	 loop ;this part of the code is executed forever
	 [
		greenon	 	
		collect-data
		wait 10 
		greenoff
	 	wait 100
	 ]
 
 end

 to collect-data
 	init-packet 42 ;create a new packet with an (abritary) packet number of 43
 	fprint readADC 6 ;read the light sensor value from ADC 6
 	fprint readADC 5 ;read the temp sensor value from ADC 5
 	fprint readADC 4 ;read the temp sensor value from ADC 4
 	add-checksum 
 end

 to yellowon
	 redon
	 greenon
 end

 to yellowoff
	 redoff
	 greenoff
 end

to calibrateADC :n 
   let [total 0]
   let [sample 10]
   repeat :sample	
   [
     let [adc (readADC :n)]
     let [total :total + :adc]
     print :adc      
     wait 10
   ]
   let [total :total / :sample]
   prs "Average:
   print :total
end
